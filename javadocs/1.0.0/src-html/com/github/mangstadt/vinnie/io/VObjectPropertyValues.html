042         * The local computer's newline character sequence.
043         */
044        private static final String NEWLINE = System.getProperty("line.separator");

046        /**
047         * <p>
048         * Unescapes all escaped characters in a property value. Escaped newlines
049         * are replaced with the local system's newline character sequence.
050         * </p>
051         * <p>
052         * <b>Example:</b>
053         * </p>
054         * 
055         * <pre class="brush:java">
056         * String value = VObjectPropertyValues.unescape(&quot;one\\,two\\;three\\nfour&quot;);
057         * assertEquals(&quot;one,two;three\nfour&quot;, value);
058         * </pre>
059         * @param value the value to unescape
060         * @return the unescaped value
061         */
062        public static String unescape(String value) {
063                return unescape(value, 0, value.length());
064        }
065
066        /**
067         * Unescapes all escaped characters in a substring.
068         * @param string the entire string
069         * @param start the start index of the substring to unescape
070         * @param end the end index (exclusive) of the substring to unescape
071         * @return the unescaped substring
072         */
073        private static String unescape(String string, int start, int end) {
074                StringBuilder sb = null;
075                boolean escaped = false;
076                for (int i = start; i < end; i++) {
077                        char c = string.charAt(i);
078
079                        if (escaped) {
080                                escaped = false;
081
082                                if (sb == null) {
083                                        sb = new StringBuilder(end - start);
084                                        sb.append(string.substring(start, i - 1));
085                                }
086
087                                switch (c) {
088                                case 'n':
089                                case 'N':
090                                        sb.append(NEWLINE);
091                                        continue;
092                                }
093
094                                sb.append(c);
095                                continue;
096                        }
097
098                        switch (c) {
099                        case '\\':
100                                escaped = true;
101                                continue;
102                        }
103
104                        if (sb != null) {
105                                sb.append(c);
106                        }
107                }
108
109                if (sb != null) {
110                        return sb.toString();
111                }
112
113                /*
114                 * The "String#substring" method makes no guarantee that the same String
115                 * object will be returned if the entire string length is passed into
116                 * the method.
117                 */
118                if (start == 0 && end == string.length()) {
119                        return string;
120                }
121
122                return string.substring(start, end);
123        }

125        /**
126         * <p>
127         * Escapes all special characters within a property value. These characters
128         * are:
129         * </p>
130         * <ul>
131         * <li>backslashes ({@code \})</li>
132         * <li>commas ({@code ,})</li>
133         * <li>semi-colons ({@code ;})</li>
134         * </ul>
135         * <p>
136         * Note: Newlines are not escaped by this method. They are escaped by
137         * {@link VObjectWriter} when the data is serialized.
138         * </p>
139         * <p>
140         * <b>Example:</b>
141         * </p>
142         * 
143         * <pre class="brush:java">
144         * String value = VObjectPropertyValues.escape(&quot;one,two;three\nfour&quot;);
145         * assertEquals(&quot;one\\,two\\;three\nfour&quot;, value);
146         * </pre>
147         * @param value the value to escape
148         * @return the escaped value
149         */
150        public static String escape(String value) {
151                StringBuilder sb = null;
152                for (int i = 0; i < value.length(); i++) {
153                        char c = value.charAt(i);
154                        switch (c) {
155                        case '\\':
156                        case ',':
157                        case ';':
158                                if (sb == null) {
159                                        sb = new StringBuilder(value.length() * 2);
160                                        sb.append(value.substring(0, i));
161                                }
162                                sb.append('\\').append(c);
163                                break;
164                        default:
165                                if (sb != null) {
166                                        sb.append(c);
167                                }
168                                break;
169                        }
170                }
171                return (sb == null) ? value : sb.toString();
172        }

174        /**
175         * Escapes all special characters within the given string.
176         * @param string the string to escape
177         * @param sb the buffer on which to append the escaped string
178         */
179        private static void escape(String string, StringBuilder sb) {
180                for (int i = 0; i < string.length(); i++) {
181                        char c = string.charAt(i);
182                        switch (c) {
183                        case '\\':
184                        case ',':
185                        case ';':
186                                sb.append('\\');
187                        }
188                        sb.append(c);
189                }
190        }

194         * Parses a "list" property value.
195         * </p>
196         * <p>
197         * List values contain multiple values separated by commas. The order that
198         * the values are in usually doesn't matter.

212                return split(value, ',', true, -1);
213        }

215        /**
216         * <p>
217         * Generates a "list" property value.
218         * </p>
219         * <p>
220         * List values contain multiple values separated by commas. The order that
221         * the values are in usually doesn't matter.
222         * </p>
223         * <p>
224         * <b>Example:</b>
225         * </p>
226         * 
227         * <pre class="brush:java">
228         * String value = VObjectPropertyValues.writeList(Arrays.asList("one", null, 2, "three,four");
229         * assertEquals("one,null,2,three\\,four", value);
230         * </pre>
231         * @param values the values to write (the {@code toString()} method is
232         * invoked on each object)
233         * @return the list value string
234         */
235        public static String writeList(Collection<?> values) {
236                StringBuilder sb = new StringBuilder();
237
238                boolean first = true;
239                for (Object value : values) {
240                        if (!first) {
241                                sb.append(',');
242                        }
243
244                        if (value == null) {
245                                sb.append("null");
246                        } else {
247                                escape(value.toString(), sb);
248                        }
249
250                        first = false;
251                }
252
253                return sb.toString();
254        }

305        /**
306         * <p>
307         * Writes a "semi-structured" property value.
308         * </p>
309         * <p>
310         * Semi-structured values contain multiple values separate by semicolons.
311         * Unlike structured values, each value cannot have their own comma
312         * separated list of sub-values. The order that the values are in matters.
313         * <p>
314         * <b>Example:</b>
315         * </p>
316         * 
317         * <pre class="brush:java">
318         * String value = VObjectPropertyValues.writeSemiStructured(Arrays.asList(&quot;one&quot;, null, 2, &quot;three;four,five&quot;, &quot;&quot;), false);
319         * assertEquals(&quot;one;null;2;three\\;four\\,five&quot;, value);
320         * </pre>
321         * @param values the values to write
322         * @param includeTrailingSemicolons true to include the semicolon delimiters
323         * for empty values at the end of the values list, false to trim them
324         * @return the semi-structured value string
325         */
326        public static String writeSemiStructured(List<?> values, boolean includeTrailingSemicolons) {
327                StringBuilder sb = new StringBuilder();
328
329                boolean first = true;
330                for (Object value : values) {
331                        if (!first) {
332                                sb.append(';');
333                        }
334
335                        if (value == null) {
336                                sb.append("null");
337                        } else {
338                                escape(value.toString(), sb);
339                        }
340
341                        first = false;
342                }
343
344                if (!includeTrailingSemicolons) {
345                        trimTrailingSemicolons(sb);
346                }
347
348                return sb.toString();
349        }

352        /**
353         * <p>
354         * Parses a "structured" property value.</p>
355         * <p>
356         * Structured values are essentially 2-D arrays. They contain multiple
357         * components separated by semicolons, and each component can have multiple
358         * values separated by commas. The order that the components are in matters,
359         * but the order that each component's list of values are in usually doesn't
360         * matter.
361         * </p>
362         * <p>
363         * <b>Example:</b>
364         * </p>
365         * 
366         * <pre class="brush:java">
367         * List&lt;List&lt;String&gt;&gt; values = VObjectPropertyValues.parseStructured("one;two,three;four\\,five\\;six");
368         * assertEquals(Arrays.asList(
369         *   Arrays.asList("one"),
370         *   Arrays.asList("two", "three"),
371         *   Arrays.asList("four,five;six")
372         * ), values);
373         * </pre>
374         * @param value the value to parse
375         * @return the parsed values
376         */
377        //@formatter:on
378        public static List<List<String>> parseStructured(String value) {
379                if (value.length() == 0) {
380                        return new ArrayList<List<String>>(0); //return a mutable list
381                }
382
383                List<List<String>> components = new ArrayList<List<String>>();
384                List<String> curComponent = new ArrayList<String>();
385                components.add(curComponent);
386
387                boolean escaped = false;
388                int cursor = 0;
389                for (int i = 0; i < value.length(); i++) {
390                        if (escaped) {
391                                escaped = false;
392                                continue;
393                        }
394
395                        char c = value.charAt(i);
396                        switch (c) {
397                        case ';':
398                                String v = unescape(value, cursor, i);
399                                if (curComponent.isEmpty() && v.length() == 0) {
400                                        /*
401                                         * If the component is empty, do not add an empty string to
402                                         * the list.
403                                         */
404                                } else {
405                                        curComponent.add(v);
406                                }
407
408                                curComponent = new ArrayList<String>();
409                                components.add(curComponent);
410                                cursor = i + 1;
411                                continue;
412                        case ',':
413                                v = unescape(value, cursor, i);
414                                curComponent.add(v);
415                                cursor = i + 1;
416                                continue;
417                        case '\\':
418                                escaped = true;
419                                continue;
420                        }
421                }
422
423                String v = unescape(value, cursor, value.length());
424                if (curComponent.isEmpty() && v.length() == 0) {
425                        /*
426                         * If the component is empty, do not add an empty string to the
427                         * list.
428                         */
429                } else {
430                        curComponent.add(v);
431                }
432
433                return components;
434        }

438         * <p>
439         * Writes a "structured" property value.
440         * </p>
441         * <p>
442         * Structured values are essentially 2-D arrays. They contain multiple
443         * components separated by semicolons, and each component can have multiple
444         * values separated by commas. The order that the components are in matters,
445         * but the order that each component's list of values are in usually doesn't
446         * matter.
447         * </p>
448         * <p>
449         * <b>Example:</b>
450         * </p>
451         * 
452         * <pre class="brush:java">
453         * String value = VObjectPropertyValues.writeStructured(Arrays.asList(
454         *   Arrays.asList("one"),
455         *   Arrays.asList("two", 3),
456         *   Arrays.asList(null),
457         *   Arrays.asList("four,five;six"),
458         *   Arrays.asList()
459         * ), false);
460         * assertEquals("one;two,3;null;four\\,five\\;six", value);
461         * </pre>
462         * @param components the components to write (each value's {@code toString()}
463         * method will be called)
464         * @param includeTrailingSemicolons true to include the semicolon delimiters
465         * for empty components at the end of the written value, false to trim them
466         * @return the structured value string
467         */
468        //@formatter:on
469        public static String writeStructured(List<? extends List<?>> components, boolean includeTrailingSemicolons) {
470                StringBuilder sb = new StringBuilder();
471                boolean firstComponent = true;
472                for (List<?> component : components) {
473                        if (!firstComponent) {
474                                sb.append(';');
475                        }
476
477                        boolean firstValue = true;
478                        for (Object value : component) {
479                                if (!firstValue) {
480                                        sb.append(',');
481                                }
482
483                                if (value == null) {
484                                        sb.append("null");
485                                } else {
486                                        escape(value.toString(), sb);
487                                }
488
489                                firstValue = false;
490                        }
491
492                        firstComponent = false;
493                }
494
495                if (!includeTrailingSemicolons) {
496                        trimTrailingSemicolons(sb);
497                }
498
499                return sb.toString();
500        }

502        /**
503         * <p>
504         * Parses a "multimap" property value.
505         * </p>
506         * <p>
507         * Multimap values are collections of key/value pairs whose keys can be
508         * multi-valued. Key/value pairs are separated by semicolons. Values are
509         * separated by commas. Keys are converted to uppercase.
510         * </p>
511         * <p>
512         * <b>Example:</b>
513         * </p>
514         * 
515         * <pre class="brush:java">
516         * Map&lt;String, List&lt;String&gt;&gt; multimap = VObjectPropertyValues.parseMultimap(&quot;one=two;THREE;FOUR=five,six\\,seven\\;eight&quot;);
517         * Map&lt;String, List&lt;String&gt;&gt; expected = new HashMap&lt;String, List&lt;String&gt;&gt;();
518         * expected.put(&quot;ONE&quot;, Arrays.asList(&quot;two&quot;));
519         * expected.put(&quot;THREE&quot;, Arrays.&lt;String&gt; asList());
520         * expected.put(&quot;FOUR&quot;, Arrays.asList(&quot;five&quot;, &quot;six,seven;eight&quot;));
521         * assertEquals(expected, multimap);
522         * </pre>
523         * @param value the value to parse
524         * @return the parsed values
525         */
526        public static Map<String, List<String>> parseMultimap(String value) {
527                if (value.length() == 0) {
528                        return new HashMap<String, List<String>>(0); //return a mutable map
529                }
530
531                Map<String, List<String>> multimap = new LinkedHashMap<String, List<String>>();
532                String curName = null;
533                List<String> curValues = new ArrayList<String>();
534
535                boolean escaped = false;
536                int cursor = 0;
537                for (int i = 0; i < value.length(); i++) {
538                        if (escaped) {
539                                escaped = false;
540                                continue;
541                        }
542
543                        char c = value.charAt(i);
544
545                        switch (c) {
546                        case ';':
547                                if (curName == null) {
548                                        curName = unescape(value, cursor, i).toUpperCase();
549                                } else {
550                                        curValues.add(unescape(value, cursor, i));
551                                }
552
553                                if (curName.length() > 0) {
554                                        if (curValues.isEmpty()) {
555                                                curValues.add("");
556                                        }
557                                        List<String> existing = multimap.get(curName);
558                                        if (existing == null) {
559                                                multimap.put(curName, curValues);
560                                        } else {
561                                                existing.addAll(curValues);
562                                        }
563                                }
564
565                                curName = null;
566                                curValues = new ArrayList<String>();
567                                cursor = i + 1;
568                                break;
569                        case '=':
570                                if (curName == null) {
571                                        curName = unescape(value, cursor, i).toUpperCase();
572                                        cursor = i + 1;
573                                }
574                                break;
575                        case ',':
576                                curValues.add(unescape(value, cursor, i));
577                                cursor = i + 1;
578                                break;
579                        case '\\':
580                                escaped = true;
581                                break;
582                        }
583                }
584
585                if (curName == null) {
586                        curName = unescape(value, cursor, value.length()).toUpperCase();
587                } else {
588                        curValues.add(unescape(value, cursor, value.length()));
589                }
590
591                if (curName.length() > 0) {
592                        if (curValues.isEmpty()) {
593                                curValues.add("");
594                        }
595                        List<String> existing = multimap.get(curName);
596                        if (existing == null) {
597                                multimap.put(curName, curValues);
598                        } else {
599                                existing.addAll(curValues);
600                        }
601                }
602
603                return multimap;
604        }

606        /**
607         * <p>
608         * Writes a "multimap" property value.
609         * </p>
610         * <p>
611         * Multimap values are collections of key/value pairs whose keys can be
612         * multi-valued. Key/value pairs are separated by semicolons. Values are
613         * separated by commas. Keys are converted to uppercase.
614         * </p>
615         * <p>
616         * <b>Example:</b>
617         * </p>
618         * 
619         * <pre class="brush:java">
620         * Map&lt;String, List&lt;Object&gt;&gt; input = new LinkedHashMap&lt;String, List&lt;Object&gt;&gt;();
621         * input.put(&quot;one&quot;, Arrays.asList(&quot;two&quot;));
622         * input.put(&quot;THREE&quot;, Arrays.&lt;String&gt; asList());
623         * input.put(&quot;FOUR&quot;, Arrays.asList(5, &quot;six,seven;eight&quot;));
624         * 
625         * String value = VObjectPropertyValues.writeMultimap(input);
626         * assertEquals(&quot;ONE=two;THREE;FOUR=5,six\\,seven\\;eight&quot;, value);
627         * </pre>
628         * @param multimap the multimap to write (each value's {@code toString()}
629         * method will be called)
630         * @return the multimap value string
631         */
632        public static String writeMultimap(Map<String, ? extends List<?>> multimap) {
633                StringBuilder sb = new StringBuilder();
634                boolean firstKey = true;
635                for (Map.Entry<String, ? extends List<?>> entry : multimap.entrySet()) {
636                        if (!firstKey) {
637                                sb.append(';');
638                        }
639
640                        String key = entry.getKey().toUpperCase();
641                        escape(key, sb);
642
643                        List<?> values = entry.getValue();
644                        if (values.isEmpty()) {
645                                continue;
646                        }
647
648                        sb.append('=');
649
650                        boolean firstValue = true;
651                        for (Object value : values) {
652                                if (!firstValue) {
653                                        sb.append(',');
654                                }
655
656                                if (value == null) {
657                                        sb.append("null");
658                                } else {
659                                        escape(value.toString(), sb);
660                                }
661
662                                firstValue = false;
663                        }
664
665                        firstKey = false;
666                }
667
668                return sb.toString();
669        }

671        /**
672         * Removes trailing semicolon characters from the end of the given buffer.
673         * @param sb the buffer
674         */
675        private static void trimTrailingSemicolons(StringBuilder sb) {
676                int index = -1;
677                for (int i = sb.length() - 1; i >= 0; i--) {
678                        char c = sb.charAt(i);
679                        if (c != ';') {
680                                index = i;
681                                break;
682                        }
683                }
684                sb.setLength(index + 1);
685        }

687        /**
688         * Splits a string.
689         * @param string the string to split
690         * @param delimiter the delimiter to escape by
691         * @param unescape true to unescape each split value, false not to
692         * @param limit the number of split values to parse or -1 to parse them all
693         * @return the split values
694         */
695        private static List<String> split(String string, char delimiter, boolean unescape, int limit) {
696                if (string.length() == 0) {
697                        return new ArrayList<String>(0); //return a mutable list
698                }
699
700                List<String> list = new ArrayList<String>();
701                boolean escaped = false;
702                int cursor = 0;
703                for (int i = 0; i < string.length(); i++) {
704                        char ch = string.charAt(i);
705
706                        if (escaped) {
707                                escaped = false;
708                                continue;
709                        }
710
711                        if (ch == delimiter) {
712                                String value = unescape ? unescape(string, cursor, i) : string.substring(cursor, i);
713                                list.add(value);
714
715                                cursor = i + 1;
716                                if (limit > 0 && list.size() == limit - 1) {
717                                        break;
718                                }
719
720                                continue;
721                        }
722
723                        switch (ch) {
724                        case '\\':
725                                escaped = true;
726                                continue;
727                        }
728                }
729
730                String value = unescape ? unescape(string, cursor, string.length()) : string.substring(cursor);
731                list.add(value);
732
733                return list;
734        }

736        /**
737         * <p>
738         * Iterates over the values in a "semi-structured" property value.
739         * </p>
740         * <p>
741         * Semi-structured values contain multiple values separate by semicolons.
742         * The order that the values are in matters.
743         * </p>
744         */
745        public static class SemiStructuredValueIterator {
746                private final Iterator<String> it;
747
748                /**
749                 * Constructs a new semi-structured value iterator.
750                 * @param value the value to parse
751                 */
752                public SemiStructuredValueIterator(String value) {
753                        this(value, -1);
754                }
755
756                /**
757                 * Constructs a new semi-structured value iterator.
758                 * @param value the value to parse
759                 * @param limit the number of values to parse, or -1 to parse all values
760                 */
761                public SemiStructuredValueIterator(String value, int limit) {
762                        it = parseSemiStructured(value, limit).iterator();
763                }
764
765                /**
766                 * Gets the next value.
767                 * @return the next value or null if the value is empty or null if there
768                 * are no more values
769                 */
770                public String next() {
771                        if (!hasNext()) {
772                                return null;
773                        }
774
775                        String next = it.next();
776                        return (next.length() == 0) ? null : next;
777                }
778
779                /**
780                 * Determines if there are any more values left.
781                 * @return true if there are more values, false if not
782                 */
783                public boolean hasNext() {
784                        return it.hasNext();
785                }
786        }

795         * </p>

834        /**
835         * <p>
836         * Iterates over the values in a "structured" property value.
837         * </p>
838         * <p>
839         * Structured values are essentially 2-D arrays. They contain multiple
840         * components separated by semicolons, and each component can have multiple
841         * values separated by commas. The order that the components are in matters,
842         * but the order that each component's list of values are in usually doesn't
843         * matter.
844         * </p>
845         */
846        public static class StructuredValueIterator {
847                private final Iterator<List<String>> it;
848
849                /**
850                 * Constructs a new structured value iterator.
851                 * @param string the structured value to parse
852                 */
853                public StructuredValueIterator(String string) {
854                        this(parseStructured(string));
855                }
856
857                /**
858                 * Constructs a new structured value iterator.
859                 * @param components the components to iterator over
860                 */
861                public StructuredValueIterator(List<List<String>> components) {
862                        it = components.iterator();
863                }
864
865                /**
866                 * Gets the first value of the next component.
867                 * @return the value or null if the component is empty or null if there
868                 * are no more components
869                 */
870                public String nextValue() {
871                        if (!hasNext()) {
872                                return null;
873                        }
874
875                        List<String> list = it.next();
876                        return list.isEmpty() ? null : list.get(0);
877                }
878
879                /**
880                 * Gets the next component.
881                 * @return the next component or an empty list of there are no more
882                 * components
883                 */
884                public List<String> nextComponent() {
885                        if (!hasNext()) {
886                                return new ArrayList<String>(0); //should be mutable
887                        }
888
889                        return it.next();
890                }
891
892                public boolean hasNext() {
893                        return it.hasNext();
894                }
895        }

905         * but the order that each component's list of values are in usually doesn't

907         * </p>

909        public static class StructuredValueBuilder {

912                /**

915                 * @param value the value
916                 * @return this
917                 */
918                public StructuredValueBuilder append(Object value) {
919                        List<Object> component = (value == null) ? Arrays.<Object> asList() : Arrays.asList(value);
920                        return append(component);

925                 * @param component the component
926                 * @return this

935                 * @return the structured value string
936                 */

939                }

944                 * delimiters for empty components at the end of the value string, false
945                 * to trim them
946                 * @return the structured value string
947                 */
