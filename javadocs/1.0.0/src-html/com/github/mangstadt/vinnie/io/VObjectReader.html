045/**
046 * <p>
047 * Parses a vobject data stream.
048 * </p>
049 * 
050 * <b>Example:</b>
051 * 
052 * <pre class="brush:java">
053 * Reader reader = ...
054 * SyntaxRules rules = SyntaxRules.vcard();
055 * VObjectReader vobjectReader = new VObjectReader(reader, rules);
056 * vobjectReader.parse(new VObjectDataListener(){ ... });
057 * vobjectReader.close();
058 * </pre>
059 * 
060 * @author Michael Angstadt
061 */
062public class VObjectReader implements Closeable {

102        /**
103         * Creates a new vobject reader.
104         * @param reader the input stream
105         * @param syntaxRules defines the rules that are used to determine what kind
106         * of syntax the data is in
107         */
108        public VObjectReader(Reader reader, SyntaxRules syntaxRules) {
109                this.reader = reader;
110                this.syntaxRules = syntaxRules;
111                stack = new ComponentStack(syntaxRules.getDefaultSyntaxStyle());
112                context = new Context(stack.names);
113
114                if (reader instanceof InputStreamReader) {
115                        InputStreamReader isr = (InputStreamReader) reader;
116                        defaultQuotedPrintableCharset = Charset.forName(isr.getEncoding());
117                } else {
118                        defaultQuotedPrintableCharset = Charset.defaultCharset();
119                }
120        }

245        /**
246         * <p>
247         * Starts or continues to parse the data off the input stream.
248         * </p>
249         * <p>
250         * This method blocks until one of the following events happen:
251         * </p>
252         * <ol>
253         * <li>The end of the input stream has been reached or</li>
254         * <li>One of the methods in the given {@link VObjectDataListener}
255         * implementation invoked {@link Context#stop()}.</li>
256         * </ol>
257         * @param listener callback interface for handling data as it is read off
258         * the input stream
259         * @throws IOException if there's a problem reading from the input stream
260         */
261        public void parse(VObjectDataListener listener) throws IOException {
262                context.stop = false;
263
264                while (!eos && !context.stop) {
265                        context.lineNumber = lineNumber;
266                        buffer.clear();
267                        context.unfoldedLine.clear();
268
269                        VObjectProperty property = parseProperty(listener);
270
271                        if (context.unfoldedLine.size() == 0) {
272                                //input stream was empty
273                                return;
274                        }
275
276                        if (property == null) {
277                                listener.onWarning(Warning.MALFORMED_LINE, null, null, context);
278                                continue;
279                        }
280
281                        /*
282                         * Note: Property names are trimmed when checking for BEGIN and END
283                         * properties because old style syntax allows there to be whitespace
284                         * around the colon character for these two properties. Component
285                         * names are trimmed for the same reason.
286                         */
287
288                        if ("BEGIN".equalsIgnoreCase(property.getName().trim())) {
289                                String componentName = property.getValue().trim().toUpperCase();
290                                if (componentName.length() == 0) {
291                                        listener.onWarning(Warning.EMPTY_BEGIN, null, null, context);
292                                        continue;
293                                }
294
295                                listener.onComponentBegin(componentName, context);
296
297                                stack.push(componentName);
298                                continue;
299                        }
300
301                        if ("END".equalsIgnoreCase(property.getName().trim())) {
302                                String componentName = property.getValue().trim().toUpperCase();
303                                if (componentName.length() == 0) {
304                                        listener.onWarning(Warning.EMPTY_END, null, null, context);
305                                        continue;
306                                }
307
308                                //find the component that this END property matches up with
309                                int popCount = stack.popCount(componentName);
310                                if (popCount == 0) {
311                                        //END property does not match up with any BEGIN properties, so ignore
312                                        listener.onWarning(Warning.UNMATCHED_END, null, null, context);
313                                        continue;
314                                }
315
316                                while (popCount > 0) {
317                                        String poppedName = stack.pop();
318                                        listener.onComponentEnd(poppedName, context);
319                                        popCount--;
320                                }
321                                continue;
322                        }
323
324                        if ("VERSION".equalsIgnoreCase(property.getName())) {
325                                String parentComponent = stack.peekName();
326                                if (syntaxRules.hasSyntaxRules(parentComponent)) {
327                                        SyntaxStyle style = syntaxRules.getSyntaxStyle(parentComponent, property.getValue());
328                                        if (style == null) {
329                                                listener.onWarning(Warning.UNKNOWN_VERSION, property, null, context);
330                                        } else {
331                                                listener.onVersion(property.getValue(), context);
332                                                stack.updateSyntax(style);
333                                                continue;
334                                        }
335                                }
336                        }
337
338                        listener.onProperty(property, context);
339                }
340        }

342        /**
343         * Parses the next property off the input stream.
344         * @param listener the data listener (for reporting warnings)
345         * @return the parsed property or null if the property could not be parsed
346         * @throws IOException if there was a problem reading from the input stream
347         */
348        private VObjectProperty parseProperty(VObjectDataListener listener) throws IOException {
349                VObjectProperty property = new VObjectProperty();
350
351                /*
352                 * The syntax style to assume the data is in.
353                 */
354                SyntaxStyle syntax = stack.peekSyntax();
355
356                /*
357                 * The name of the parameter we're currently inside of.
358                 */
359                String curParamName = null;
360
361                /*
362                 * The character that was used to escape the current character (for
363                 * parameter values).
364                 */
365                char paramValueEscapeChar = 0;
366
367                /*
368                 * Are we currently inside a parameter value that is surrounded with
369                 * double-quotes?
370                 */
371                boolean inQuotes = false;
372
373                /*
374                 * Are we currently inside the property value?
375                 */
376                boolean inValue = false;
377
378                /*
379                 * Does the line use quoted-printable encoding, and does it end all of
380                 * its folded lines with a "=" character?
381                 */
382                boolean foldedQuotedPrintableLine = false;
383
384                /*
385                 * Are we currently inside the whitespace that prepends a folded line?
386                 */
387                boolean inFoldedLineWhitespace = false;
388
389                /*
390                 * The current character.
391                 */
392                char ch = 0;
393
394                /*
395                 * The previous character.
396                 */
397                char prevChar;
398
399                while (true) {
400                        prevChar = ch;
401
402                        int read = nextChar();
403                        if (read < 0) {
404                                //end of stream
405                                eos = true;
406                                break;
407                        }
408
409                        ch = (char) read;
410
411                        if (prevChar == '\r' && ch == '\n') {
412                                /*
413                                 * The newline was already processed when the "\r" character was
414                                 * encountered, so ignore the accompanying "\n" character.
415                                 */
416                                continue;
417                        }
418
419                        if (isNewline(ch)) {
420                                foldedQuotedPrintableLine = (inValue && prevChar == '=' && property.getParameters().isQuotedPrintable());
421                                if (foldedQuotedPrintableLine) {
422                                        /*
423                                         * Remove the "=" character that some vCards put at the end
424                                         * of quoted-printable lines that are followed by a folded
425                                         * line.
426                                         */
427                                        buffer.chop();
428                                        context.unfoldedLine.chop();
429                                }
430
431                                //keep track of the current line number
432                                lineNumber++;
433
434                                continue;
435                        }
436
437                        if (isNewline(prevChar)) {
438                                if (isWhitespace(ch)) {
439                                        /*
440                                         * This line is a continuation of the previous line (the
441                                         * line is folded).
442                                         */
443                                        inFoldedLineWhitespace = true;
444                                        continue;
445                                }
446
447                                if (foldedQuotedPrintableLine) {
448                                        /*
449                                         * The property's parameters indicate that the property
450                                         * value is quoted-printable. And the previous line ended
451                                         * with an equals sign. This means that folding whitespace
452                                         * may not be prepended to folded lines like it should.
453                                         */
454                                } else {
455                                        /*
456                                         * We're reached the end of the property.
457                                         */
458                                        leftOver = ch;
459                                        break;
460                                }
461                        }
462
463                        if (inFoldedLineWhitespace) {
464                                if (isWhitespace(ch) && syntax == SyntaxStyle.OLD) {
465                                        /*
466                                         * 2.1 allows multiple whitespace characters to be used for
467                                         * folding (section 2.1.3).
468                                         */
469                                        continue;
470                                }
471                                inFoldedLineWhitespace = false;
472                        }
473
474                        context.unfoldedLine.append(ch);
475
476                        if (inValue) {
477                                buffer.append(ch);
478                                continue;
479                        }
480
481                        //decode escaped parameter value character
482                        if (paramValueEscapeChar != 0) {
483                                char escapeChar = paramValueEscapeChar;
484                                paramValueEscapeChar = 0;
485
486                                switch (escapeChar) {
487                                case '\\':
488                                        switch (ch) {
489                                        case '\\':
490                                                buffer.append(ch);
491                                                continue;
492                                        case ';':
493                                                /*
494                                                 * Semicolons can only be escaped in old style parameter
495                                                 * values. If a new style parameter value has
496                                                 * semicolons, the value should be surrounded in double
497                                                 * quotes.
498                                                 */
499                                                buffer.append(ch);
500                                                continue;
501                                        }
502                                        break;
503                                case '^':
504                                        switch (ch) {
505                                        case '^':
506                                                buffer.append(ch);
507                                                continue;
508                                        case 'n':
509                                                buffer.append(NEWLINE);
510                                                continue;
511                                        case '\'':
512                                                buffer.append('"');
513                                                continue;
514                                        }
515                                        break;
516                                }
517
518                                /*
519                                 * Treat the escape character as a normal character because it's
520                                 * not a valid escape sequence.
521                                 */
522                                buffer.append(escapeChar).append(ch);
523                                continue;
524                        }
525
526                        //check for a parameter value escape character
527                        if (curParamName != null) {
528                                switch (syntax) {
529                                case OLD:
530                                        if (ch == '\\') {
531                                                paramValueEscapeChar = ch;
532                                                continue;
533                                        }
534                                        break;
535                                case NEW:
536                                        if (ch == '^' && caretDecodingEnabled) {
537                                                paramValueEscapeChar = ch;
538                                                continue;
539                                        }
540                                        break;
541                                }
542                        }
543
544                        //set the group
545                        if (ch == '.' && property.getGroup() == null && property.getName() == null) {
546                                property.setGroup(buffer.getAndClear());
547                                continue;
548                        }
549
550                        if ((ch == ';' || ch == ':') && !inQuotes) {
551                                if (property.getName() == null) {
552                                        //set the property name
553                                        property.setName(buffer.getAndClear());
554                                } else {
555                                        //set a parameter value
556                                        String paramValue = buffer.getAndClear();
557                                        if (syntax == SyntaxStyle.OLD) {
558                                                //old style allows whitespace to surround the "=", so remove it
559                                                paramValue = ltrim(paramValue);
560                                        }
561                                        property.getParameters().put(curParamName, paramValue);
562                                        curParamName = null;
563                                }
564
565                                if (ch == ':') {
566                                        //the rest of the line is the property value
567                                        inValue = true;
568                                }
569                                continue;
570                        }
571
572                        if (property.getName() != null) {
573                                //it's a multi-valued parameter
574                                if (ch == ',' && curParamName != null && !inQuotes && syntax != SyntaxStyle.OLD) {
575                                        String paramValue = buffer.getAndClear();
576                                        property.getParameters().put(curParamName, paramValue);
577                                        continue;
578                                }
579
580                                //set the parameter name
581                                if (ch == '=' && curParamName == null) {
582                                        String paramName = buffer.getAndClear().toUpperCase();
583                                        if (syntax == SyntaxStyle.OLD) {
584                                                //old style allows whitespace to surround the "=", so remove it
585                                                paramName = rtrim(paramName);
586                                        }
587                                        curParamName = paramName;
588                                        continue;
589                                }
590
591                                //entering/leaving a double-quoted parameter value (new style only)
592                                if (ch == '"' && curParamName != null && syntax != SyntaxStyle.OLD) {
593                                        inQuotes = !inQuotes;
594                                        continue;
595                                }
596                        }
597
598                        buffer.append(ch);
599                }
600
601                /*
602                 * Line or stream ended before the property value was reached.
603                 */
604                if (!inValue) {
605                        return null;
606                }
607
608                property.setValue(buffer.getAndClear());
609                if (property.getParameters().isQuotedPrintable()) {
610                        decodeQuotedPrintable(property, listener);
611                }
612
613                return property;
614        }

616        /**
617         * Decodes the given property's value from quoted-printable encoding.
618         * @param property the property
619         * @param listener the data listener
620         */
621        private void decodeQuotedPrintable(VObjectProperty property, VObjectDataListener listener) {
622                Charset charset = getCharset(property, listener);
623                if (charset == null) {
624                        charset = defaultQuotedPrintableCharset;
625                }
626
627                String value = property.getValue();
628                QuotedPrintableCodec codec = new QuotedPrintableCodec(charset.name());
629                try {
630                        value = codec.decode(value);
631                } catch (DecoderException e) {
632                        listener.onWarning(Warning.QUOTED_PRINTABLE_ERROR, property, e, context);
633                        return;
634                }
635
636                property.setValue(value);
637        }

639        /**
640         * Gets the character set the given property is encoded in.
641         * @param property the property
642         * @param listener the data listener
643         * @return the character set or null if the character is not set or could
644         * not be determined
645         */
646        private Charset getCharset(VObjectProperty property, VObjectDataListener listener) {
647                Exception thrown;
648                try {
649                        return property.getParameters().getCharset();
650                } catch (IllegalCharsetNameException e) {
651                        //name contains illegal characters
652                        thrown = e;
653                } catch (UnsupportedCharsetException e) {
654                        //not recognized by the JVM
655                        thrown = e;
656                }
657
658                listener.onWarning(Warning.UNKNOWN_CHARSET, property, thrown, context);
659                return null;
660        }

662        /**
663         * Gets the next character in the input stream.
664         * @return the next character or -1 if the end of the stream has been
665         * reached
666         * @throws IOException if there's a problem reading from the input stream
667         */
668        private int nextChar() throws IOException {
669                if (leftOver >= 0) {
670                        /*
671                         * Use the character that was left over from the previous invocation
672                         * of "readLine()".
673                         */
674                        int ch = leftOver;
675                        leftOver = -1;
676                        return ch;
677                }
678
679                return reader.read();
680        }

682        /**
683         * Determines if the given character is a newline character.
684         * @param ch the character
685         * @return true if it's a newline character, false if not
686         */
687        private static boolean isNewline(char ch) {
688                return ch == '\n' || ch == '\r';
689        }
690
691        /**
692         * Determines if the given character is a space or a tab.
693         * @param ch the character
694         * @return true if it's a space or a tab, false if not
695         */
696        private static boolean isWhitespace(char ch) {
697                return ch == ' ' || ch == '\t';
698        }

700        /**
701         * Keeps track of the hierarchy of nested components and their syntax
702         * styles.
703         */
704        private static class ComponentStack {
705                /**
706                 * The hierarchy of components the parser is currently inside of.
707                 */
708                private final List<String> names = new ArrayList<String>();
709
710                /**
711                 * <p>
712                 * The syntax style of each component in the hierarchy.
713                 * </p>
714                 * 
715                 * <p>
716                 * Note: This will always be one element larger than the "names" list
717                 * because it must remember the style of the "root" (for properties that
718                 * are not inside of a component, should there happen to be any).
719                 * </p>
720                 */
721                private final List<SyntaxStyle> syntax = new ArrayList<SyntaxStyle>();
722
723                /**
724                 * Creates a new stack.
725                 * @param defaultSyntax the default syntax style
726                 */
727                public ComponentStack(SyntaxStyle defaultSyntax) {
728                        syntax.add(defaultSyntax);
729                }
730
731                /**
732                 * Pushes a component onto the stack.
733                 * @param component the component name
734                 */
735                public void push(String component) {
736                        names.add(component);
737                        syntax.add(peekSyntax());
738                }
739
740                /**
741                 * Removes the top component from the stack.
742                 * @return the name of the component that was removed
743                 */
744                public String pop() {
745                        syntax.remove(syntax.size() - 1);
746                        return names.remove(names.size() - 1);
747                }
748
749                /**
750                 * Gets the number of calls to {@link #pop()} it would take to pop the
751                 * given component name.
752                 * @param name the component name
753                 * @return the number of pops or 0 if the name was not found
754                 */
755                public int popCount(String name) {
756                        int index = names.lastIndexOf(name);
757                        return (index < 0) ? 0 : names.size() - index;
758                }
759
760                /**
761                 * Gets the top component name.
762                 * @return the top component name or null if the name stack is empty
763                 */
764                public String peekName() {
765                        return names.isEmpty() ? null : names.get(names.size() - 1);
766                }
767
768                /**
769                 * Gets the top syntax style.
770                 * @return the top syntax style or null if the syntax stack is empty
771                 */
772                public SyntaxStyle peekSyntax() {
773                        return syntax.isEmpty() ? null : syntax.get(syntax.size() - 1);
774                }
775
776                /**
777                 * Replaces the top syntax style.
778                 * @param style the syntax style
779                 */
780                public void updateSyntax(SyntaxStyle style) {
