025package com.github.mangstadt.vinnie.io;
026
027import com.github.mangstadt.vinnie.VObjectProperty;
028
029/**
030 * Callback interface used by {@link VObjectReader} for handling data as it is
031 * parsed off the data stream.
032 * @see VObjectReader#parse
033 * @author Michael Angstadt
034 */
035public interface VObjectDataListener {
036        /**
037         * Called when a component begins (i.e. when a BEGIN property is
038         * encountered).
039         * @param name the component name (in uppercase)
040         * @param context the parse context
041         */
042        void onComponentBegin(String name, Context context);
043
044        /**
045         * <p>
046         * Called when a component ends (i.e. when an END property is encountered).
047         * </p>
048         * <p>
049         * If a component ends before any of its subcomponents end, then the parser
050         * will generate "onComponentEnd" events for each subcomponent.
051         * </p>
052         * 
053         * <p>
054         * For example, the following data will cause the following sequence of
055         * events to occur (notice that the outer component, A, ends before its
056         * subcomponents):
057         * </p>
058         * 
059         * <p>
060         * <b>Data:</b>
061         * </p>
062         * 
063         * <pre class="brush:java">
064         * BEGIN:A
065         * BEGIN:B
066         * BEGIN:C
067         * END:A
068         * END:C
069         * END:B
070         * </pre>
071         * <p>
072         * <b>Sequence of events:</b>
073         * </p>
074         * <ol>
075         * <li>onComponentBegin("A", [])</li>
076         * <li>onComponentBegin("B", ["A"])</li>
077         * <li>onComponentBegin("C", ["A", "B"])</li>
078         * <li>onComponentEnd("C", ["A", "B"])</li>
079         * <li>onComponentEnd("B", ["A"])</li>
080         * <li>onComponentEnd("A", [])</li>
081         * <li>onWarning(UNMATCHED_END)</li>
082         * <li>onWarning(UNMATCHED_END)</li>
083         * </ol>
084         * @param name the component name (in uppercase)
085         * @param context the parse context
086         */
087        void onComponentEnd(String name, Context context);
088
089        /**
090         * Called when a property is read.
091         * @param property the property
092         * @param context the parse context
093         */
094        void onProperty(VObjectProperty property, Context context);
095
096        /**
097         * Called when a VERSION property is read whose value and position (as
098         * defined in {@link SyntaxRules}) are recognized as valid.
099         * @param value the version string
100         * @param context the parse context
101         */
102        void onVersion(String value, Context context);
103
104        /**
105         * Called when a non-fatal error occurs during parsing.
106         * @param warning the warning
107         * @param property the property that the warning is associated with, or null
108         * if the warning is not associated with a property
109         * @param thrown the exception that was thrown or null if no exception was
110         * thrown
111         * @param context the parse context
112         */
113        void onWarning(Warning warning, VObjectProperty property, Exception thrown, Context context);
114}
